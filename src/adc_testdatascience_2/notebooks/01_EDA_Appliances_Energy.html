# 🧹 Clean Data
def clean_data(df):
    print("\n🧹 Cleaning data: removing NaNs and zero values...")
    df = df.dropna()
    df = df[df['Appliances'] > 0]
    df = df.drop(columns=['rv1', 'rv2'], errors='ignore')
    print(f"✅ Cleaned dataset shape: {df.shape}")
    return df


df_cleaned = clean_data(df)

# 🔁 Analyze Seasonality of Target Variable
def analyze_seasonality(df, column='Appliances'):
    print("\n📈 Analyzing seasonality...")
    result = seasonal_decompose(df[column], model='additive', period=1440)
    result.plot()
    plt.suptitle("Seasonal Decomposition", fontsize=14)
    plt.tight_layout()
    plt.show()

    adf_statistic, adf_p_value = adfuller(df[column])[:2]
    print(f"ADF Statistic: {adf_statistic:.4f}")
    print(f"p-value for ADF Test: {adf_p_value:.4f}")
    print("❗ The series is " + ("stationary" if adf_p_value < 0.05 else "non-stationary"))

    seasonality_strength = np.var(result.seasonal) / np.var(result.observed)
    if seasonality_strength > 0.8:
        print("🌞 Seasonality strength: Very high")
    elif seasonality_strength > 0.5:
        print("🌤️ Seasonality strength: Moderate")
    elif seasonality_strength > 0.2:
        print("🌥️ Seasonality strength: Mild")
    else:
        print("🌑 Seasonality strength: Low or negligible")

    plot_acf(df[column], lags=50)
    plt.title("Autocorrelation - Appliances")
    plt.show()

analyze_seasonality(df_cleaned)

# 📈 Rolling Mean Plots
def plot_rolling_means(df, column='Appliances'):
    plt.figure(figsize=(12, 6))
    df[column].plot(label='Original', alpha=0.5)
    for window in [6*6, 6*24, 7*24*6]:
        df[column].rolling(window=window).mean().plot(label=f'Rolling mean (window={window})')
    plt.title("Rolling Means of 'Appliances'")
    plt.legend()
    plt.grid(True)
    plt.tight_layout()
    plt.show()

plot_rolling_means(df_cleaned)
